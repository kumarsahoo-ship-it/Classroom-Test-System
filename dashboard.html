function loadLeaderboard(testName) { fetch(`https://classroom-test-system.onrender.com/leaderboard/${testName}`) .then(res => { if (!res.ok) { throw new Error("API not found"); } return res.json(); }) .then(data => { const table = document.getElementById("leaderboardTable"); table.innerHTML = ` Rank Student Score Total `; if (data.length === 0) { table.innerHTML += `No results yet`; return; } data.forEach((r, index) => { table.innerHTML += ` ${index + 1} ${r.username} ${r.score} ${r.total} `; }); }) .catch(err => { console.log("Leaderboard error:", err); alert("Leaderboard API not working"); }); }
